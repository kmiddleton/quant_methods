Distribution of P values

set.seed(3210)

nn <- 10
group1.mean <- 6
group2.mean <- 6

niter <- 100000
ps <- numeric(length = niter)

for(ii in 1:niter) {
  y <- c(rnorm(nn, group1.mean, 1), rnorm(nn, group2.mean, 1))
  gg <- rep(c('a','b'), each = nn)
  ps[ii] <- t.test(y ~ gg, var.equal = TRUE)$p.value
}
